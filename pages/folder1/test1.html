/**
 * Promise е®һзҺ° йҒөеҫӘpromise/A+и§„иҢғ
 * Promise/A+и§„иҢғиҜ‘ж–Ү:
 * https://malcolmyu.github.io/2015/06/12/Promises-A-Plus/#note-4
 */

// promise дёүдёӘзҠ¶жҖҒ
const PENDING = "pending";
const FULFILLED = "fulfilled";
const REJECTED = "rejected";

function Promise(excutor) {
    let that = this; // зј“еӯҳеҪ“еүҚpromiseе®һдҫӢеҜ№иұЎ
    that.status = PENDING; // еҲқе§ӢзҠ¶жҖҒ
    that.value = undefined; // fulfilledзҠ¶жҖҒж—¶ иҝ”еӣһзҡ„дҝЎжҒҜ
    that.reason = undefined; // rejectedзҠ¶жҖҒж—¶ жӢ’з»қзҡ„еҺҹеӣ 
    that.onFulfilledCallbacks = []; // еӯҳеӮЁfulfilledзҠ¶жҖҒеҜ№еә”зҡ„onFulfilledеҮҪж•°
    that.onRejectedCallbacks = []; // еӯҳеӮЁrejectedзҠ¶жҖҒеҜ№еә”зҡ„onRejectedеҮҪж•°

    function resolve(value) { // valueжҲҗеҠҹжҖҒж—¶жҺҘж”¶зҡ„з»ҲеҖј
        if(value instanceof Promise) {
            return value.then(resolve, reject);
        }

        // дёәд»Җд№Ҳresolve еҠ setTimeout?
        // 2.2.4и§„иҢғ onFulfilled е’Ң onRejected еҸӘе…Ғи®ёеңЁ execution context ж Ҳд»…еҢ…еҗ«е№іеҸ°д»Јз Ғж—¶иҝҗиЎҢ.

        setTimeout(() => {
            // и°ғз”Ёresolve еӣһи°ғеҜ№еә”onFulfilledеҮҪж•°
            if (that.status === PENDING) {
                // еҸӘиғҪз”ұpendingзҠ¶жҖҒ => fulfilledзҠ¶жҖҒ (йҒҝе…Қи°ғз”ЁеӨҡж¬Ўresolve reject)
                that.status = FULFILLED;
                that.value = value;
                that.onFulfilledCallbacks.forEach(cb => cb(that.value));
            }
        });
    }

    function reject(reason) { // reasonеӨұиҙҘжҖҒж—¶жҺҘж”¶зҡ„жӢ’еӣ 
        setTimeout(() => {
            // и°ғз”Ёreject еӣһи°ғеҜ№еә”onRejectedеҮҪж•°
            if (that.status === PENDING) {
                // еҸӘиғҪз”ұpendingзҠ¶жҖҒ => rejectedзҠ¶жҖҒ (йҒҝе…Қи°ғз”ЁеӨҡж¬Ўresolve reject)
                that.status = REJECTED;
                that.reason = reason;
                that.onRejectedCallbacks.forEach(cb => cb(that.reason));
            }
        });
    }

    // жҚ•иҺ·еңЁexcutorжү§иЎҢеҷЁдёӯжҠӣеҮәзҡ„ејӮеёё
    // new Promise((resolve, reject) => {
    //     throw new Error('error in excutor')
    // })
    try {
        excutor(resolve, reject);
    } catch (e) {
        reject(e);
    }
}
