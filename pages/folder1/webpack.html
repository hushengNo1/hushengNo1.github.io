# 简介

  • webpack4.0开始支持零配置
  • webpack基于nodejs开发，所以导入、导出规则都基于commonJs规范。module.exports = {},require()

# webpack-dev-server

1、overlay参数：当存在编译错误或警告时，在浏览器中显示全屏覆盖。默认情况下禁用。如果你只想显示编译错误:

devServer: {
  overlay: {
    warnings: true,
    errors: true
  }
}

# webpack中处理CSS

module.exports = {
  module: {
    rules: [
      {
        test: /\.css$/,
        use: ["style-loader", "css-loader", "postcss-loader"]
      }
    ]
  }
}
//自动加前缀
//And create a postcss.config.js with:
module.exports = {
  plugins: [
    require('autoprefixer')
  ]
}
//或者
module.exports = {
  plugins: {
    autoprefixer: {}
  }
}

  • mini-css-extract-plugin：把css抽离成一个文件

  • optimize-css-assets-webpack-plugin：压缩css

    const OptimizeCSSAssetsWebpackPlugin = require('optimize-css-assets-webpack-plugin');
const UglifyJsWebpackPlugin = require('uglifyjs-webpack-plugin');

module.exports = {
  //配置优化规则
  optimization: {
    //压缩优化
    minimizer: [
      //压缩css，(产生问题：就是压缩不再执行自己默认的压缩方式了，也走这个插件，导致无法压缩，故需要指定js的压缩方式：uglifyjs-webpack-plugin)
      new OptimizeCSSAssetsWebpackPlugin(),
      //压缩js---->需要在module中配置js的编译规则
      new UglifyJsWebpackPlugin({
        cache:true,//是否使用缓存
        parallel:true,//是否是并发编译（可以同时编译多个js）
        sourceMap:true,//启动源码映射（方便调试）
      }),
    ],
  },
  module: {
    rules: [
      {
        test: /\.js$/,
        exclude: /node_modules/,
        loader: "babel-loader",
        options: {
          //转换的语法预设
          presets: ["stage-3"]
        }
      }
    ]
  },
};
