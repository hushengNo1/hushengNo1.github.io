# 1.从连接获取参数

  1. function param2Obj(url) {
  console.log('encodeURIComponent--', encodeURIComponent(url));
  console.log('decodeURIComponent--', decodeURIComponent(url.split('?')[1]));
  const search = decodeURIComponent(url.split('?')[1]);
  if (!search) {
    return {};
  }
  const obj = {};
  const searchArr = search.split('&');
  searchArr.forEach(v => {
    const index = v.indexOf('=');
    if (index !== -1) {
      const name = v.substring(0, index);
      const val = v.substring(index + 1, v.length);
      obj[name] = val;
    }
  });
  console.log('obj--', obj);
  return obj;
}
param2Obj('http://www.hah.com/pp?hu=胡+胜');
